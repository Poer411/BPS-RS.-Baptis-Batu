async function prosesKirimWA(data) { // 1. DAFTAR KONTAK TEKNISI (Update Otomatis) const daftarTeknisi = { "DIDIT": "6281331755209", "SUSANTO": "6289668467140", "LUCIA": "6285708203992", "ROKIM": "6285806690875", "BAMBANG": "628980506696", "PURNOMO": "6285100332530", "TWIN": "6285646405019" }; // Ambil nama teknisi dari database (pastikan huruf besar/kecil sama dengan daftar di atas) const namaAdminPilih = data.ditugaskan_ke ? data.ditugaskan_ke.toUpperCase() : ""; // Tentukan Nomor Tujuan (Jika nama tidak terdaftar, kirim ke nomor PURNOMO/Admin sebagai backup) const nomorTujuan = daftarTeknisi[namaAdminPilih] || "6285100332530"; // 2. Cegah Tampilan "null" agar rapi const namaAlat = data.nama_alat || "Tidak disebutkan"; const lokasiAlat = data.lokasi || "Cek Lokasi"; const masalah = data.isi_laporan || "Mohon cek unit segera"; const urgensi = data.urgensi || "Normal"; const namaTeknisi = data.ditugaskan_ke || "Petugas IPSRS"; updateLog(`🔔 Trigger! Mengirim ke ${namaTeknisi} (${nomorTujuan})...`); const pesan = `🎯 *TUGAS BARU IPSRS*\n\n👤 *Teknisi:* ${namaTeknisi}\n⚙️ *Alat:* ${namaAlat}\n📍 *Lokasi:* ${lokasiAlat}\n📝 *Masalah:* ${masalah}\n⚠️ *Urgensi:* ${urgensi}`; try { const response = await fetch('https://api.fonnte.com/send', { method: 'POST', headers: { 'Authorization': FONNTE_TOKEN }, body: new URLSearchParams({ 'target': nomorTujuan, 'message': pesan }) }); const result = await response.json(); if (result.status === true) { updateLog(`✅ Berhasil Terkirim ke ${namaTeknisi} (${nomorTujuan})`); // Reset status kirim_wa agar tidak looping await db.from('laporan_perbaikan').update({ kirim_wa: false }).eq('id', data.id); updateLog(`♻️ Sistem Standby Kembali.`); } else { updateLog(`❌ Fonnte Error: ${result.reason}`); } } catch (err) { updateLog(`❌ Gagal: ${err.message}`); } }
